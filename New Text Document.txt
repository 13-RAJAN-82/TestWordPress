client.on('ready', () => {
    setInterval(() => {
        let membersCount = client.guilds.cache.map(guild => guild.memberCount).reduce((a, b) => a + b, 0)
        client.user.setActivity(`[${membersCount}] Players`, {type: "PLAYING"});
    }, 1000 * 60);

});
